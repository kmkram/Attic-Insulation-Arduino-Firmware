#include <Wire.h>

#define MMA8452_ADDR 0x1C    // Try 0x1D if not found

#define STATUS      0x00
#define OUT_X_MSB   0x01
#define WHO_AM_I    0x0D
#define XYZ_DATA_CFG 0x0E
#define CTRL_REG1   0x2A

void writeRegister(byte reg, byte value)
{
  Wire.beginTransmission(MMA8452_ADDR);
  Wire.write(reg);
  Wire.write(value);
  Wire.endTransmission();
}

byte readRegister(byte reg)
{
  Wire.beginTransmission(MMA8452_ADDR);
  Wire.write(reg);
  Wire.endTransmission(false);

  Wire.requestFrom(MMA8452_ADDR, 1);
  return Wire.read();
}

void readAccel(float &xg, float &yg, float &zg)
{
  Wire.beginTransmission(MMA8452_ADDR);
  Wire.write(OUT_X_MSB);
  Wire.endTransmission(false);

  Wire.requestFrom(MMA8452_ADDR, 6);

  int16_t x = ((Wire.read() << 8) | Wire.read()) >> 4;
  int16_t y = ((Wire.read() << 8) | Wire.read()) >> 4;
  int16_t z = ((Wire.read() << 8) | Wire.read()) >> 4;

  // sign extend 12-bit values
  if (x > 2047) x -= 4096;
  if (y > 2047) y -= 4096;
  if (z > 2047) z -= 4096;

  // ±2g mode: 1024 counts/g
  xg = x / 1024.0;
  yg = y / 1024.0;
  zg = z / 1024.0;
}

void setup()
{
  Serial.begin(115200);
  Wire.begin();

  delay(100);

  byte id = readRegister(WHO_AM_I);

  Serial.print("WHO_AM_I = 0x");
  Serial.println(id, HEX);

  if (id != 0x2A)
  {
    Serial.println("Not an MMA8452Q or wrong I2C address.");
    while (1);
  }

  // Put in standby
  writeRegister(CTRL_REG1, 0x00);

  // ±2g range
  writeRegister(XYZ_DATA_CFG, 0x00);

  // Active mode, 100 Hz
  writeRegister(CTRL_REG1, 0x19);

  Serial.println("MMA8452Q initialized");
}

void loop()
{
  float x, y, z;

  readAccel(x, y, z);

  Serial.print("X=");
  Serial.print(x, 3);

  Serial.print("g  Y=");
  Serial.print(y, 3);

  Serial.print("g  Z=");
  Serial.print(z, 3);

  Serial.println("g");

  delay(250);
}
